module.exports.couponDetail = async function(_id){
  const coupon = await db.coupon.aggregate([
    {
      $match: { _id }
    }, {
      // shop 조인
      $lookup: {
        from: 'shop',
        localField: 'shopId',
        foreignField: '_id',
        as: 'shop'
      }
    }, {
      // shop 조인 결과(배열)를 낱개의 속성으로 변환한다.
      $unwind: '$shop'
    }, {
      // epilogue 조인
      $lookup: {
        from: 'epilogue',
        localField: '_id',
        foreignField: 'couponId',
        as: 'epilogueList'
      }
    },
  ]).next();
  return coupon;
};
